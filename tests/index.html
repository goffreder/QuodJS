}, {

    return 10;
  },

  }
});

  foo : function() {
    return Super(this).foo() * 2;
  }
}).Extends('A');

  foo : function() {
    return Super(this).foo() * 2;
  }
}).Extends('B');


var a = New('A');
var b = New('B');
var c = New('C');
